server {                                                                                                                           
listen 443 ssl;                                                                                                                
server_name 192.168.3.43;                                                                                                                                                                                                                                    
# SSL Configuration                                                                                                            
ssl_certificate /etc/ssl/certs/helloworld.crt;
ssl_certificate_key /etc/ssl/private/helloworld.key;
ssl_protocols TLSv1.2 TLSv1.3;                                                                                                 
ssl_ciphers 'ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384:ECDHE-ECDSA-CHACHA20-POLY1305:ECDHE-RSA-CHACHA20-PO>    
    ssl_prefer_server_ciphers on;
    ssl_dhparam /etc/nginx/dhparam.pem;
    # ssl_stapling on;
    # ssl_stapling_verify on;
    # resolver 8.8.8.8 8.8.4.4 valid=300s;
    # resolver_timeout 5s;

    # HTTP Security Headers
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    add_header X-Content-Type-Options nosniff;
    add_header X-Frame-Options DENY;
    add_header X-XSS-Protection "1; mode=block";
    add_header Referrer-Policy "no-referrer-when-downgrade";
    add_header Content-Security-Policy "
        default-src 'self';
        connect-src 'self' https://192.168.3.43:8089;
        script-src 'self';
        style-src 'self' 'unsafe-inline' https://cdn.jsdelivr.net;
        font-src 'self' https://fonts.gstatic.com;
        img-src 'self' data:;
        object-src 'none';
        base-uri 'self';
        form-action 'self';
        " always;

    # Logs
    access_log /var/log/nginx/access.log;
    error_log /var/log/nginx/error.log warn;

    # Buffer Overflow
    client_body_buffer_size 10K;
    client_header_buffer_size 1k;
    client_max_body_size 1M;
    large_client_header_buffers 2 1k;

    # Timeout
    client_body_timeout 12;
    client_header_timeout 12;
    send_timeout 10;
    keepalive_timeout 15;

    # Root Directory and Index
    root /var/www/helloworld;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;
        limit_req zone=one burst=10 nodelay;
        limit_conn addr 10;
    }
}

server {
    listen 80;
    server_name 192.168.3.43;

    # Redirect HTTP to HTTPS
    return 301 https://$server_name$request_uri;
}

# Rate Limiting
limit_req_zone $binary_remote_addr zone=one:10m rate=5r/s;
limit_conn_zone $binary_remote_addr zone=addr:10m;
